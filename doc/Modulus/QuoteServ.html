204:       $log.debug "QuoteServ", "Got: #{origin.raw}"
205: 
206:       if origin.argsArr.length < 1
207:         Modulus.reply(origin, "Usage: ADD quotation")

210: 
211:       user = Modulus.users.find(origin.source)
212: 
213:       return if user == nil
214: 
215:       unless user.logged_in?
216:         Modulus.reply(origin, "You must be logged in to a services account in order to use this command.")
217:         return
218:       end
219: 
220:       unless origin.args.length < 300
221:         Modulus.reply(origin, "Quotations cannot be longer than 300 characters.")
222:         return
223:       end
224: 
225:       unless Modulus.link.isChannel? origin.target
226:         source = "Private Message"
227:       else
228:         source = origin.target
229:       end
230: 
231:       Quote.create(
232:         :content => origin.args,
233:         :place_added => source,
234:         :date_added => DateTime.now(),
235:         :added_by_nick => origin.source,
236:         :account_name => user.svid
237:       )
238: 
239:       Modulus.reply(origin, "Quotation successfully added to the database.")
240:       $log.info 'QuoteServ', "#{origin.source} added a new quotation to the database from #{source}"

247:         Modulus.reply(origin, "Usage: SEARCH content")
248:         return
249:       end
250:

252: 
253:       return if user == nil
254:

256:         Modulus.reply(origin, "You must be logged in to a services account in order to use this command.")
257:         return
258:       end
259: 
260:       found = Quote.find(:all, :conditions => ['content like ?', "%#{origin.args}%"])
261: 
262:       if found.count == 0
263:         Modulus.reply(origin, "No matches.")
264:       elsif found.count > 3
265:         Modulus.reply(origin, "Found #{found.count} matches.#{" Displaying first 3." if found.count > 3}")
266:         pos = 0
267: 
268:         found.each { |result|
269:           self.show_quotation(origin, result)
270:           pos += 1
271: 
272:           break if pos == 3
273:         }
274:       else
275:         found.each { |result|
276:           self.show_quotation(origin, result)
277:         }
278:       end
