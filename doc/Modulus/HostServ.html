132:     def cmd_hs_set(origin)
133:       $log.debug "HostServ", "Got: #{origin.raw}"
134: 
135:       if origin.argsArr.length != 1
136:         Modulus.reply(origin, "Usage: SET hostname")
137:         return
138:       end
139: 
140:       user = Modulus.users.find(origin.source)
141:       return if user == nil
142: 
143:       account = Account.find_by_username(user.svid)
144: 
145:       if account == nil
146:         Modulus.reply(origin, "You must be logged in to a valid services account to use this command.")
147:         return
148:       end
149: 
150:       hostname = origin.argsArr[0]
151: 
152:       # TODO: Make this the widest range possible depending on what the server supports.
153:       unless hostname.match(/[a-zA-Z0-9.-]/)
154:         Modulus.reply(origin, "That host name is not valid. Host names may only contain alphanumeric characters, '-' and '.'")
155:         return
156:       end
157: 
158:       unless user.is_services_admin?
159:         restricted = Modulus.config.getOption("HostServ", "restricted_hostnames").split(" ")
160: 
161:         restricted.each { |restr|
162:           restr.gsub!("*", ".*")
163: 
164:           if hostname.match(restr)
165:             Modulus.reply(origin, "The host name you provided is not permitted on this network.")
166:             return
167:           end
168:         }
169: 
170:         approval = Modulus.config.getBool("HostServ", "oper_approval")
171: 
172:       else
173:         approval = false
174:       end
175: 
176:       oldHost = Host.find_by_account_id(account.id)
177: 
178:       if oldHost == nil
179:         Host.create(
180:           :account_id => account.id,
181:           :date_added => DateTime.now,
182:           :hostname => hostname,
183:           :approved => !approval)
184:       else
185:         oldHost.hostname = hostname
186:         oldHost.date_added = DateTime.now
187:         oldHost.save!
188:         oldHost.approved = !approval
189:       end
190: 
191:       if approval
192:         Modulus.reply(origin, "Your host name has been submitted for approval. When approved by network staff, it will be automatically activated.")
193:         $log.info 'HostServ', "Action required: #{origin.source} has requested host name: #{hostname}"
194:       else
195:         self.activate(origin.source, hostname)
196: 
197:         Modulus.reply(origin, "The host name you provided has been saved and activated.")
198:         $log.info 'HostServ', "#{origin.source} has requested and activated host name: #{hostname}"
199:       end
200:     end
